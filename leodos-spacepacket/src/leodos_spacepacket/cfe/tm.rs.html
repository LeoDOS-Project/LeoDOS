3//! CFE-specific telemetry packet definitions and builder.
4
5use crate::{
6    Apid, PacketSequenceCount, PacketType, PrimaryHeader, SecondaryHeaderFlag, SequenceFlag,
7    SpacePacketData, builder::Vacant, cfe::tm_builder::TelemetryBuilder,
8};
9use core::mem::size_of;
10use zerocopy::{FromBytes, Immutable, IntoBytes, KnownLayout, Unaligned};
11
12/// The CFE telemetry secondary header (6-byte time + 4-byte padding).
13#[repr(C)]
14#[derive(IntoBytes, FromBytes, Unaligned, KnownLayout, Immutable, Default, Copy, Clone, Debug)]
15pub struct TelemetrySecondaryHeader {
16    /// 6-byte CCSDS Day Segmented (CDS) time format.
17    pub time: [u8; 6],
18    /// Padding to ensure the payload that follows is 64-bit aligned.
19    pub spare: [u8; 4],
20}
21
22/// A zero-copy view over a complete CFE telemetry packet (headers + payload).
23#[repr(C)]
24#[derive(Debug, FromBytes, IntoBytes, Unaligned, KnownLayout, Immutable)]
25pub struct Telemetry<P: SpacePacketData> {
26    pub primary: PrimaryHeader,
27    pub secondary: TelemetrySecondaryHeader,
28    pub payload: P,
29}
30
31/// An error that can occur when building a CFE telemetry packet.
32#[derive(Debug, Copy, Clone, Eq, PartialEq)]
33pub enum TelemetryError {
34    BufferTooSmall { required: usize, provided: usize },
35}
36
37impl Telemetry<()> {
38    /// Creates a new telemetry packet builder.
39    pub fn builder() -> TelemetryBuilder<Vacant, Vacant, Vacant> {
40        TelemetryBuilder::new()
41    }
42}
43
44impl<P: SpacePacketData> Telemetry<P> {
45    /// Creates a new telemetry packet view over the provided buffer.
46    pub fn new<'a>(
47        buffer: &'a mut [u8],
48        apid: Apid,
49        sequence_count: PacketSequenceCount,
50        time: [u8; 6],
51        payload: &P,
52    ) -> Result<&'a mut Telemetry<P>, TelemetryError> {
53        let total_size = size_of::<Telemetry<P>>();
54        if buffer.len() < total_size {
55            return Err(TelemetryError::BufferTooSmall {
56                required: total_size,
57                provided: buffer.len(),
58            });
59        }
60
61        let tlm_packet = Telemetry::<P>::mut_from_bytes(&mut buffer[..total_size])
62            .expect("should not fail due to size check");
63
64        tlm_packet
65            .primary
66            .set_version(crate::PacketVersion::VERSION_1);
67        tlm_packet.primary.set_packet_type(PacketType::Telemetry);
68        tlm_packet
69            .primary
70            .set_secondary_header_flag(SecondaryHeaderFlag::Present);
71        tlm_packet.primary.set_apid(apid);
72        tlm_packet
73            .primary
74            .set_sequence_flag(SequenceFlag::Unsegmented);
75        tlm_packet.primary.set_sequence_count(sequence_count);
76
77        let data_len = (size_of::<TelemetrySecondaryHeader>() + size_of::<P>()) as u16;
78        tlm_packet.primary.set_data_field_len(data_len);
79
80        tlm_packet.secondary.time = time;
81
82        tlm_packet
83            .payload
84            .as_mut_bytes()
85            .copy_from_slice(payload.as_bytes());
86
87        Ok(tlm_packet)
88    }
89}
