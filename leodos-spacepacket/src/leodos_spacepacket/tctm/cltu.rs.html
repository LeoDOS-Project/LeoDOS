1//! Generates a Communications Link Transmission Unit (CLTU), the final data unit
2//! sent to a satellite's radio for uplink.
3//!
4//! A CLTU is a highly robust data structure that wraps a TC Transfer Frame.
5//! It adds error-correction coding and special sequences to ensure reliable
6//! reception of commands by the spacecraft.
7
8const START_SEQUENCE: &[u8] = &[0xEB, 0x90];
9const TAIL_SEQUENCE: &[u8] = &[0xC5; 8];
10
11/// An error that can occur during CLTU encoding.
12#[derive(Debug, Copy, Clone, Eq, PartialEq)]
13pub enum CltuError {
14    /// The provided output buffer is too small to hold the encoded CLTU.
15    OutputBufferTooSmall { required: usize, provided: usize },
16}
17
18/// Computes the required buffer size for encoding a TC frame of a given length into a CLTU.
19///
20/// This is a helper function to allow users to allocate a correctly-sized buffer
21/// before calling `encode_cltu`.
22pub fn encoded_cltu_len(tc_frame_len: usize) -> usize {
23    let num_blocks = (tc_frame_len + 6) / 7; // Ceiling division
24    START_SEQUENCE.len() + num_blocks * 8 + TAIL_SEQUENCE.len()
25}
26
27/// Encodes a TC Transfer Frame byte slice into a CLTU in the provided buffer.
28///
29/// This function performs two critical operations for uplink reliability:
30/// 1.  It applies a (63, 56) Bose-Chaudhuri-Hocquhem (BCH) forward error-correction code,
31///     which allows the spacecraft to automatically correct bit errors.
32/// 2.  It wraps the encoded data with a standard Start Sequence and Tail Sequence to
33///     ensure the spacecraft's radio can reliably detect the beginning and end of the command.
34///
35/// The provided `tc_frame_bytes` should typically be randomized before calling this function.
36///
37/// Returns the total number of bytes written to the output buffer.
38pub fn encode_cltu(tc_frame_bytes: &[u8], output_buffer: &mut [u8]) -> Result<usize, CltuError> {
39    let required_len = encoded_cltu_len(tc_frame_bytes.len());
40    if output_buffer.len() < required_len {
41        return Err(CltuError::OutputBufferTooSmall {
42            required: required_len,
43            provided: output_buffer.len(),
44        });
45    }
46
47    let mut writer_idx = 0;
48
49    // Write Start Sequence
50    output_buffer[writer_idx..writer_idx + START_SEQUENCE.len()].copy_from_slice(START_SEQUENCE);
51    writer_idx += START_SEQUENCE.len();
52
53    // Write BCH-encoded blocks
54    for chunk in tc_frame_bytes.chunks(7) {
55        let mut block = [0x55; 7]; // Pad with alternating 01010101 pattern
56        block[..chunk.len()].copy_from_slice(chunk);
57
58        let parity = bch::compute_bch_parity(&block);
59
60        output_buffer[writer_idx..writer_idx + 7].copy_from_slice(&block);
61        writer_idx += 7;
62        output_buffer[writer_idx] = parity;
63        writer_idx += 1;
64    }
65
66    // Write Tail Sequence
67    output_buffer[writer_idx..writer_idx + TAIL_SEQUENCE.len()].copy_from_slice(TAIL_SEQUENCE);
68    writer_idx += TAIL_SEQUENCE.len();
69
70    Ok(writer_idx)
71}

73mod bch {
74    const fn generate_lookup_table() -> [u8; 256] {
75        const CCSDS_POLYNOMIAL: u8 = 0b1011_0101; // x^7 + x^6 + x^4 + x^2 + 1 (transposed)
76        let mut table = [0u8; 256];
77        let mut i = 0;
78        while i < 256 {
79            let mut val = i as u8;
80            let mut bit = 0;
81            while bit < 8 {
82                val = if val & 1 == 1 {
83                    (val >> 1) ^ CCSDS_POLYNOMIAL
84                } else {
85                    val >> 1
86                };
87                bit += 1;
88            }
89            table[i] = val;
90            i += 1;
91        }
92        table
93    }
94
95    const LOOKUP_TABLE: [u8; 256] = generate_lookup_table();
96
97    /// Computes the (63, 56) BCH codeword as defined in CCSDS 231.0-B-4.
98    pub fn compute_bch_parity(bytes: &[u8; 7]) -> u8 {
99        let remainder = bytes
100            .iter()
101            .fold(0, |acc, &val| LOOKUP_TABLE[(acc ^ val) as usize]);
102        !remainder & 0x7F
103    }
104}

106#[cfg(test)]
107mod tests {
108    use super::*;
109
110    #[test]
111    fn bch_test_vectors() {
112        assert_eq!(
113            bch::compute_bch_parity(&[0x22, 0xF6, 0x00, 0xFF, 0x00, 0x42, 0x1A]),
114            0x4A
115        );
116        assert_eq!(
117            bch::compute_bch_parity(&[0x8C, 0xC0, 0x0E, 0x01, 0x0D, 0x19, 0x06]),
118            0x16
119        );
120    }
121
122    #[test]
123    fn cltu_encoding() {
124        let tc_frame: &[u8] = &[0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08];
125        let mut cltu_buffer = [0u8; 34]; // 2 (start) + 2*8 (blocks) + 8 (tail) = 26 bytes needed
126
127        let len = encode_cltu(tc_frame, &mut cltu_buffer).unwrap();
128        assert_eq!(len, 26);
129
130        let expected_start = &cltu_buffer[0..2];
131        assert_eq!(expected_start, &[0xEB, 0x90]);
132
133        // First block: 01,02,03,04,05,06,07 -> Parity 0x1F
134        assert_eq!(
135            cltu_buffer[2..9],
136            [0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07]
137        );
138        assert_eq!(cltu_buffer[9], 0x1F);
139
140        // Second block: 08,55,55,55,55,55,55 -> Parity 0x47
141        assert_eq!(
142            cltu_buffer[10..17],
143            [0x08, 0x55, 0x55, 0x55, 0x55, 0x55, 0x55]
144        );
145        assert_eq!(cltu_buffer[17], 0x47);
146
147        let expected_tail = &cltu_buffer[18..26];
148        assert_eq!(expected_tail, &[0xC5; 8]);
149    }
