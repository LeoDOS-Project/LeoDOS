16/// A typed builder for defining the specification of a `SpacePacket`.
17#[derive(Clone)]
18pub struct SpacePacketBuilder<A, B, C, D, E, F> {
19    pub(crate) apid: A,
20    pub(crate) packet_type: B,
21    pub(crate) sequence_count: C,
22    pub(crate) secondary_header_flag: SecondaryHeaderFlag,
23    pub(crate) sequence_flag: SequenceFlag,
24    pub(crate) buffer: D,
25    pub(crate) data_field_len: E,
26    pub(crate) crc_alg: F,
27}
28
29/// An error that occurs during the building of a `SpacePacket`.
30#[derive(Copy, Clone, Eq, PartialEq, Hash, Debug)]
31pub enum BuilderError {
32    Spec(BuildError),
33    BufferTooSmall { required: usize, provided: usize },
34}
35
36impl core::fmt::Display for BuilderError {
37    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
38        match self {
39            BuilderError::Spec(e) => write!(f, "Specification error: {e}"),
40            BuilderError::BufferTooSmall { required, provided } => {
41                write!(
42                    f,
43                    "Buffer too small: required {required}, provided {provided}"
44                )
45            }
46        }
47    }
48}
49
50impl From<BuildError> for BuilderError {
51    fn from(e: BuildError) -> Self {
52        BuilderError::Spec(e)
53    }
54}

180impl<'a> SpacePacketBuilder<Apid, PacketType, PacketSequenceCount, &'a mut [u8], u16, Vacant> {
181    /// Consumes the writer and builds the final `SpacePacket` view.
182    pub fn build(self) -> Result<&'a mut SpacePacket, BuilderError> {
183        let sp = SpacePacket::new(
184            self.buffer,
185            self.apid,
186            self.packet_type,
187            self.sequence_count,
188            self.secondary_header_flag,
189            self.sequence_flag,
190            self.data_field_len,
191        )?;
192        Ok(sp)
193    }
194}

196#[cfg(test)]
197mod tests {
198    use super::*;
199    use zerocopy::IntoBytes;
200    use zerocopy::byteorder::network_endian::U32;
201
202    #[test]
203    fn new_builder_happy_path() {
204        let mut buffer = [0u8; 100];
205        let payload = U32::new(0xDEADBEEF);
206
207        let packet = SpacePacket::builder()
208            .apid(Apid::IDLE)
209            .packet_type(PacketType::Telemetry)
210            .sequence_count(PacketSequenceCount::new())
211            .buffer(&mut buffer)
212            .data_type::<U32>()
213            .build()
214            .unwrap();
215
216        packet.set_data_field(&payload).unwrap();
217
218        assert_eq!(packet.data_field(), payload.as_bytes());
219        assert_eq!(packet.data_field_len(), size_of::<U32>());
220        assert_eq!(packet.apid(), Apid::IDLE);
221    }
