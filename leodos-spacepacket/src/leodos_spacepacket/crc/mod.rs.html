18/// An error related to CRC operations.
19#[derive(Copy, Clone, Eq, PartialEq, Hash, Debug)]
20pub enum CrcError {
21    /// An error occurred during the underlying packet build.
22    Build(BuilderError),
23    /// The buffer was too small to hold the packet and its CRC.
24    BufferTooSmall { required: usize, provided: usize },
25    /// The calculated CRC did not match the one in the buffer.
26    ValidationFailed { expected: u16, calculated: u16 },
27    /// An error occurred parsing the underlying data field.
28    DataField(crate::DataFieldError),
29}
30
31impl core::fmt::Display for CrcError {
32    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
33        match self {
34            CrcError::Build(e) => write!(f, "Build error: {e}"),
35            CrcError::BufferTooSmall { required, provided } => {
36                write!(
37                    f,
38                    "Buffer too small for CRC packet: required {required}, provided {provided}"
39                )
40            }
41            CrcError::ValidationFailed {
42                expected,
43                calculated,
44            } => {
45                write!(
46                    f,
47                    "CRC validation failed: expected {expected:#06X}, calculated {calculated:#06X}"
48                )
49            }
50            CrcError::DataField(e) => write!(f, "Data field error: {e}"),
51        }
52    }
53}
54
55impl From<BuilderError> for CrcError {
56    fn from(e: BuilderError) -> Self {
57        CrcError::Build(e)
58    }
59}
60impl From<crate::DataFieldError> for CrcError {
61    fn from(e: crate::DataFieldError) -> Self {
62        CrcError::DataField(e)
63    }
64}

66/// A wrapper around a `SpacePacket` that automatically manages
67/// a trailing CRC-16 checksum.
68///
69/// This struct is created by the builder's `.crc().build()` method. All data
70/// access through this wrapper is CRC-protected.
71pub struct CrcSpacePacket<'a, 'b> {
72    packet: &'a mut SpacePacket,
73    crc_bytes: &'a mut [u8],
74    crc_alg: &'b crc::Crc<u16>,
75}
76
77impl<'a, 'b> CrcSpacePacket<'a, 'b> {
78    pub fn new(
79        buffer: &'a mut [u8],
80        apid: Apid,
81        packet_type: PacketType,
82        sequence_count: PacketSequenceCount,
83        secondary_header_flag: crate::SecondaryHeaderFlag,
84        sequence_flag: crate::SequenceFlag,
85        data_field_len: u16,
86        crc_alg: &'b crc::Crc<u16>,
87    ) -> Result<CrcSpacePacket<'a, 'b>, CrcError> {
88        let required_len = size_of::<PrimaryHeader>() + data_field_len as usize;
89        if required_len + 2 > buffer.len() {
90            return Err(CrcError::BufferTooSmall {
91                required: required_len + 2,
92                provided: buffer.len(),
93            });
94        }
95
96        let (packet_buf, crc_buf) = buffer[..required_len + 2].split_at_mut(required_len);
97        let packet = SpacePacket::mut_from_bytes(packet_buf).unwrap();
98
99        // Build the header
100        packet.set_version(PacketVersion::VERSION_1);
101        packet.set_packet_type(packet_type);
102        packet.set_apid(apid);
103        packet.set_sequence_count(sequence_count);
104        packet.set_data_field_len(data_field_len);
105        packet.set_secondary_header_flag(secondary_header_flag);
106        packet.set_sequence_flag(sequence_flag);
107
108        // Create the wrapper
109        let mut crc_packet = CrcSpacePacket {
110            packet,
111            crc_bytes: crc_buf,
112            crc_alg: crc_alg,
113        };
114
115        // Set the initial CRC
116        crc_packet.update_crc();
117
118        Ok(crc_packet)
119    }
120    /// Writes data to the packet's data field and automatically updates the CRC.
121    ///
122    /// This is the safe, CRC-aware way to set the packet's payload.
123    pub fn set_data<T: SpacePacketData>(&mut self, data: &T) -> Result<(), crate::DataFieldError> {
124        self.packet.set_data_field(data)?;
125        self.update_crc();
126        Ok(())
127    }
128
129    /// Validates the CRC and returns a typed, zero-copy view of the data field.
130    pub fn data_as<T: SpacePacketData>(&self) -> Result<&T, CrcError> {
131        self.validate()?;
132        Ok(self.packet.data_as::<T>()?)
133    }
134
135    /// Validates the CRC and returns an immutable slice of the data field.
136    pub fn data(&self) -> Result<&[u8], CrcError> {
137        self.validate()?;
138        Ok(self.packet.data_field())
139    }
140
141    /// Validates the current CRC against the packet's contents.
142    pub fn validate(&self) -> Result<(), CrcError> {
143        let expected = U16::read_from_bytes(self.crc_bytes).unwrap().get();
144        let calculated = self.crc_alg.checksum(self.packet.as_bytes());
145        if expected == calculated {
146            Ok(())
147        } else {
148            Err(CrcError::ValidationFailed {
149                expected,
150                calculated,
151            })
152        }
153    }
154
155    /// Forces a recalculation and update of the CRC value.
156    /// This is called automatically by `set_data_field`.
157    pub fn update_crc(&mut self) {
158        let calculated = self.crc_alg.checksum(self.packet.as_bytes());
159        U16::new(calculated)
160            .write_to_prefix(self.crc_bytes)
161            .unwrap();
162    }
163}
