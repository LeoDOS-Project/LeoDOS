704/// Retrieves the symbolic name for an OSAL status code.
705pub fn get_error_name(error_num: i32) -> Result<CString<{ ffi::OS_ERROR_NAME_LENGTH as usize }>> {
706    let mut name_buf = [0 as libc::c_char; ffi::OS_ERROR_NAME_LENGTH as usize];
707    check(unsafe { ffi::OS_GetErrorName(error_num, &mut name_buf) })?;
708
709    // The result from OS_GetErrorName is null-terminated.
710    let c_str = unsafe { core::ffi::CStr::from_ptr(name_buf.as_ptr()) };
711    let mut s = CString::new();
712    s.extend_from_bytes(c_str.to_bytes())
713        .map_err(|_| Error::OsErrNameTooLong)?;
714    Ok(s)
715}

733    status_code: i32,
734) -> Result<CString<{ ffi::OS_STATUS_STRING_LENGTH as usize }>> {
735    let mut name_buf = [0 as libc::c_char; ffi::OS_STATUS_STRING_LENGTH as usize];
736    unsafe { ffi::OS_StatusToString(status_code, &mut name_buf) };
737
738    let c_str = unsafe { core::ffi::CStr::from_ptr(name_buf.as_ptr()) };
739    let mut s = CString::new();
740    s.extend_from_bytes(c_str.to_bytes())
741        .map_err(|_| Error::OsErrNameTooLong)?;
