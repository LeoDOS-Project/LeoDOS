1//! Safe, idiomatic wrappers for OSAL Task APIs.
2//!
3//! This module provides utilities for task management, such as delaying the
4//! current task, retrieving task IDs, and managing task priorities.
5
6use crate::error::{Error, Result};
7use crate::ffi;
8use crate::os::id::OsalId;
9use crate::status::check;
10use core::ffi::CStr;
11use core::mem::MaybeUninit;
12use core::time::Duration;
13use heapless::CString;

15/// A type-safe, zero-cost wrapper for an OSAL Task ID.
16#[derive(Debug, Clone, Copy, PartialEq, Eq)]
17#[repr(transparent)]
18pub struct TaskId(pub ffi::osal_id_t);
19
20impl TaskId {
21    /// Sets the priority of this task.
22    ///
23    /// # Arguments
24    /// * `priority`: The new priority (0=highest, 255=lowest).
25    pub fn set_priority(&self, priority: u8) -> Result<()> {
26        check(unsafe { ffi::OS_TaskSetPriority(self.0, priority) })?;
27        Ok(())
28    }
29
30    /// Retrieves the `TaskId` of the currently executing task.
31    pub fn current() -> Result<Self> {
32        let id = unsafe { ffi::OS_TaskGetId() };
33        if id == 0 {
34            // OS_TaskGetId returns 0 if not called from a valid task context.
35            Err(Error::OsErrInvalidId)
36        } else {
37            Ok(Self(id))
38        }
39    }
40
41    /// Finds an existing OSAL task ID by its name.
42    pub fn from_name(name: &str) -> Result<Self> {
43        let mut c_name = CString::<{ ffi::OS_MAX_API_NAME as usize }>::new();
44        c_name
45            .extend_from_bytes(name.as_bytes())
46            .map_err(|_| Error::OsErrNameTooLong)?;
47
48        let mut task_id = MaybeUninit::uninit();
49        check(unsafe { ffi::OS_TaskGetIdByName(task_id.as_mut_ptr(), c_name.as_ptr()) })?;
50        Ok(Self(unsafe { task_id.assume_init() }))
51    }
52}

54/// Type alias for a handler function called on task deletion.
55pub type TaskDeleteHandler = unsafe extern "C" fn();

57/// Properties of an OSAL task, returned by `Task::get_info`.
58#[derive(Debug, Clone)]
59pub struct TaskProp {
60    /// The registered name of the task.
61    pub name: CString<{ ffi::OS_MAX_API_NAME as usize }>,
62    /// The OSAL ID of the task that created this task.
63    pub creator: OsalId,
64    /// The allocated stack size of the task in bytes.
65    pub stack_size: usize,
66    /// The priority of the task (0=highest, 255=lowest).
67    pub priority: u8,
68}

70/// A handle to an OSAL task.
71///
72/// This is a wrapper around an `osal_id_t` that will automatically call
73/// `OS_TaskDelete` when it goes out of scope, preventing resource leaks.
74#[derive(Debug)]
75pub struct Task {
76    id: TaskId,
77}
78
79impl Task {
80    /// Creates a new OSAL task and starts it running.
81    ///
82    /// The stack for the task is allocated from the system memory heap.
83    ///
84    /// # Arguments
85    /// * `name`: A unique string to identify the task.
86    /// * `entry_point`: The function that the new task will execute. This must
87    ///   be a free function with `extern "C"` linkage.
88    /// * `stack_size`: The size of the stack to allocate for the new task.
89    /// * `priority`: The priority of the new task (0=highest, 255=lowest).
90    pub fn new(
91        name: &str,
92        entry_point: unsafe extern "C" fn(),
93        stack_size: usize,
94        priority: u8,
95    ) -> Result<Self> {
96        let mut c_name = CString::<{ ffi::OS_MAX_API_NAME as usize }>::new();
97        c_name
98            .extend_from_bytes(name.as_bytes())
99            .map_err(|_| Error::OsErrNameTooLong)?;
100
101        let mut task_id = MaybeUninit::uninit();
102        check(unsafe {
103            ffi::OS_TaskCreate(
104                task_id.as_mut_ptr(),
105                c_name.as_ptr(),
106                Some(entry_point),
107                core::ptr::null_mut(), // OSAL_TASK_STACK_ALLOCATE
108                stack_size,
109                priority,
110                0, // flags
111            )
112        })?;
113
114        Ok(Self {
115            id: TaskId(unsafe { task_id.assume_init() }),
116        })
117    }
118
119    /// Returns the underlying `TaskId` for this task.
120    pub fn id(&self) -> TaskId {
121        self.id
122    }
123
124    /// Retrieves information about this task.
125    pub fn get_info(&self) -> Result<TaskProp> {
126        let mut prop = MaybeUninit::<ffi::OS_task_prop_t>::uninit();
127        check(unsafe { ffi::OS_TaskGetInfo(self.id.0, prop.as_mut_ptr()) })?;
128        let prop = unsafe { prop.assume_init() };
129
130        let mut name_str = CString::new();
131        let c_str = unsafe { CStr::from_ptr(prop.name.as_ptr()) };
132        name_str
133            .extend_from_bytes(c_str.to_bytes())
134            .map_err(|_| Error::OsErrNameTooLong)?;
135
136        Ok(TaskProp {
137            name: name_str,
138            creator: OsalId(prop.creator),
139            stack_size: prop.stack_size,
140            priority: prop.priority,
141        })
142    }
143}

145impl Drop for Task {
146    /// Deletes the OSAL task when the `Task` object goes out of scope.
147    fn drop(&mut self) {
148        let _ = unsafe { ffi::OS_TaskDelete(self.id.0) };
149    }
150}

152/// Delays the execution of the current task for at least the specified duration.
153///
154/// # Arguments
155/// * `duration`: The minimum amount of time to delay.
156pub fn delay(duration: Duration) -> Result<()> {
157    let millis = duration.as_millis();
158    // Clamp to u32::MAX, which is ~49 days. This is a reasonable upper limit for a delay.
159    let millis_u32 = millis.try_into().unwrap_or(u32::MAX);
160    check(unsafe { ffi::OS_TaskDelay(millis_u32) })?;
161    Ok(())
162}

164/// Installs a handler function to be called when the current task is deleted.
165///
166/// This is useful for cleaning up resources that a task creates before it is
167/// removed from the system.
168///
169/// # Safety
170/// The provided `handler` function must be a valid `extern "C"` function pointer.
171/// It will be called in the context of task deletion, so it should be brief
172/// and avoid complex operations or blocking, especially any that would try to
173/// interact further with OSAL.
174pub fn install_delete_handler(handler: TaskDeleteHandler) -> Result<()> {
175    check(unsafe { ffi::OS_TaskInstallDeleteHandler(Some(handler)) })?;
176    Ok(())
177}
178
179/// Exits the calling task.
180///
181/// This function terminates the currently running task and does not return.
182pub fn exit() -> ! {
183    unsafe {
184        ffi::OS_TaskExit();
185    }
186    // This function never returns, but we add a loop to satisfy the compiler.
187    loop {}
188}

190/// Reverse-looks up an OSAL task ID from an underlying operating system ID.
191///
192/// This is for special cases like exception handling where the OS provides a
193/// native task ID that needs to be mapped back to an OSAL ID.
194///
195/// # Safety
196/// The `sys_data` pointer must be a valid pointer to the OS-specific task identifier data,
197/// and `sys_data_size` must be the correct size of that data.
198pub unsafe fn find_id_by_system_data(sys_data: *const (), sys_data_size: usize) -> Result<TaskId> {
199    let mut task_id = MaybeUninit::uninit();
200    check(ffi::OS_TaskFindIdBySystemData(
201        task_id.as_mut_ptr(),
202        sys_data as *const _,
203        sys_data_size,
204    ))?;
205    Ok(TaskId(task_id.assume_init()))
206}
