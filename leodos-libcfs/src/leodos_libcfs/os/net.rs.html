16/// Defines how to shut down a TCP stream.
17#[derive(Debug, Clone, Copy, PartialEq, Eq)]
18#[repr(u32)]
19pub enum SocketShutdownMode {
20    /// Disable future reading.
21    Read = ffi::OS_SocketShutdownMode_t_OS_SocketShutdownMode_SHUT_READ,
22    /// Disable future writing.
23    Write = ffi::OS_SocketShutdownMode_t_OS_SocketShutdownMode_SHUT_WRITE,
24    /// Disable future reading and writing.
25    ReadWrite = ffi::OS_SocketShutdownMode_t_OS_SocketShutdownMode_SHUT_READWRITE,
26}

28impl SocketAddr {
29    /// Creates a new IPv4 socket address.
30    pub fn new_ipv4(ip_addr: &str, port: u16) -> Result<Self> {
31        let mut addr_uninit = MaybeUninit::uninit();
32
33        // 1. Initialize for the correct domain
34        let status = unsafe {
35            ffi::OS_SocketAddrInit(
36                addr_uninit.as_mut_ptr(),
37                ffi::OS_SocketDomain_t_OS_SocketDomain_INET,
38            )
39        };
40        check(status)?;
41
42        // 2. Set the IP address from a string
43        let mut c_ip: String<64> = String::new();
44        c_ip.push_str(ip_addr)
45            .map_err(|_| Error::OsErrNameTooLong)?;
46        c_ip.push('\0').map_err(|_| Error::OsErrNameTooLong)?;
47        let status = unsafe {
48            ffi::OS_SocketAddrFromString(
49                addr_uninit.as_mut_ptr(),
50                c_ip.as_ptr() as *const libc::c_char,
51            )
52        };
53        check(status)?;
54
55        // 3. Set the port
56        let status = unsafe { ffi::OS_SocketAddrSetPort(addr_uninit.as_mut_ptr(), port) };
57        check(status)?;
58
59        Ok(Self(unsafe { addr_uninit.assume_init() }))
60    }
61
62    /// Gets the port number from a socket address.
63    pub fn port(&self) -> Result<u16> {
64        let mut port = MaybeUninit::uninit();
65        check(unsafe { ffi::OS_SocketAddrGetPort(port.as_mut_ptr(), &self.0) })?;
66        Ok(unsafe { port.assume_init() })
67    }
68
69    /// Gets a string representation of the host address (e.g., "127.0.0.1").
70    pub fn to_string(&self) -> Result<String<64>> {
71        let mut buffer = [0u8; 64];
72        check(unsafe {
73            ffi::OS_SocketAddrToString(
74                buffer.as_mut_ptr() as *mut libc::c_char,
75                buffer.len(),
76                &self.0,
77            )
78        })?;
79        let len = buffer.iter().position(|&b| b == 0).unwrap_or(buffer.len());
80        let vec = heapless::Vec::from_slice(&buffer[..len]).map_err(|_| Error::OsErrNameTooLong)?;
81        let s = String::from_utf8(vec).map_err(|_| Error::InvalidString)?;
82        Ok(s)
83    }
84}

86/// A raw socket handle that ensures `OS_close` is called on drop.
87#[derive(Debug)]
88#[repr(transparent)]
89pub struct Socket(ffi::osal_id_t);
90
91impl Drop for Socket {
92    fn drop(&mut self) {
93        if self.0 != 0 {
94            let _ = unsafe { ffi::OS_close(self.0) };
95        }
96    }
97}

99impl Socket {
100    /// Binds a socket to a given local address.
101    pub fn bind_address(&self, addr: &SocketAddr) -> Result<()> {
102        check(unsafe { ffi::OS_SocketBindAddress(self.0, &addr.0) })?;
103        Ok(())
104    }
105}

107/// A UDP socket.
108#[derive(Debug)]
109#[repr(transparent)]
110pub struct UdpSocket(Socket);
111
112impl UdpSocket {
113    /// Creates a new UDP socket bound to the specified address.
114    pub fn bind(addr: SocketAddr) -> Result<UdpSocket> {
115        let mut sock_id = MaybeUninit::uninit();
116        check(unsafe {
117            ffi::OS_SocketOpen(
118                sock_id.as_mut_ptr(),
119                ffi::OS_SocketDomain_t_OS_SocketDomain_INET,
120                ffi::OS_SocketType_t_OS_SocketType_DATAGRAM,
121            )
122        })?;
123        let sock_id = unsafe { sock_id.assume_init() };
124
125        check(unsafe { ffi::OS_SocketBind(sock_id, &addr.0) })?;
126
127        Ok(UdpSocket(Socket(sock_id)))
128    }
129
130    /// Receives a single datagram message on the socket.
131    pub fn recv_from<'a>(&self, buf: &'a mut [u8]) -> Result<(usize, SocketAddr)> {
132        let mut remote_addr_uninit = MaybeUninit::<ffi::OS_SockAddr_t>::uninit();
133
134        let num_bytes = unsafe {
135            ffi::OS_SocketRecvFrom(
136                (self.0).0,
137                buf.as_mut_ptr() as *mut _,
138                buf.len(),
139                remote_addr_uninit.as_mut_ptr(),
140                ffi::OS_PEND as i32,
141            )
142        };
143
144        if num_bytes >= 0 {
145            Err(Error::from(num_bytes))
146        } else {
147            let remote_addr = unsafe { remote_addr_uninit.assume_init() };
148            Ok((num_bytes as usize, SocketAddr(remote_addr)))
149        }
150    }
151
152    /// Sends data on the socket to the given address.
153    pub fn send_to(&self, buf: &[u8], target: &SocketAddr) -> Result<usize> {
154        let num_bytes = unsafe {
155            ffi::OS_SocketSendTo(self.0 .0, buf.as_ptr() as *const _, buf.len(), &target.0)
156        };
157        if num_bytes < 0 {
158            Err(Error::from(num_bytes))
159        } else {
160            Ok(num_bytes as usize)
161        }
162    }
163
164    /// Receives a datagram with an absolute timeout.
165    pub fn recv_from_abs<'a>(
166        &self,
167        buf: &'a mut [u8],
168        abstime: OsTime,
169    ) -> Result<(usize, SocketAddr)> {
170        let mut remote_addr_uninit = MaybeUninit::<ffi::OS_SockAddr_t>::uninit();
171        let num_bytes = unsafe {
172            ffi::OS_SocketRecvFromAbs(
173                self.0 .0,
174                buf.as_mut_ptr() as *mut _,
175                buf.len(),
176                remote_addr_uninit.as_mut_ptr(),
177                abstime.0,
178            )
179        };
180
181        if num_bytes < 0 {
182            Err(Error::from(num_bytes))
183        } else {
184            let remote_addr = SocketAddr(unsafe { remote_addr_uninit.assume_init() });
185            Ok((num_bytes as usize, remote_addr))
186        }
187    }
188}

190/// A TCP socket server, listening for connections.
191#[derive(Debug)]
192#[repr(transparent)]
193pub struct TcpListener(Socket);
194
195impl TcpListener {
196    /// Creates a new `TcpListener` which will be bound to the specified address.
197    pub fn bind(addr: SocketAddr) -> Result<Self> {
198        let mut sock_id = MaybeUninit::uninit();
199        check(unsafe {
200            ffi::OS_SocketOpen(
201                sock_id.as_mut_ptr(),
202                ffi::OS_SocketDomain_t_OS_SocketDomain_INET,
203                ffi::OS_SocketType_t_OS_SocketType_STREAM,
204            )
205        })?;
206        let sock_id = unsafe { sock_id.assume_init() };
207
208        check(unsafe { ffi::OS_SocketBind(sock_id, &addr.0) })?;
209
210        Ok(TcpListener(Socket(sock_id)))
211    }
212
213    /// Accepts a new incoming connection from this listener.
214    /// This function will block until a new connection is established.
215    pub fn accept(&self) -> Result<(TcpStream, SocketAddr)> {
216        let mut remote_addr_uninit = MaybeUninit::<ffi::OS_SockAddr_t>::uninit();
217        let mut conn_sock_id = MaybeUninit::uninit();
218        check(unsafe {
219            ffi::OS_SocketAccept(
220                self.0 .0,
221                conn_sock_id.as_mut_ptr(),
222                remote_addr_uninit.as_mut_ptr(),
223                ffi::OS_PEND as i32,
224            )
225        })?;
226
227        let stream = TcpStream(Socket(unsafe { conn_sock_id.assume_init() }));
228        let remote_addr = SocketAddr(unsafe { remote_addr_uninit.assume_init() });
229        Ok((stream, remote_addr))
230    }
231
232    /// Places the socket into a listening state for incoming connections.
233    ///
234    /// This is typically called after `bind`.
235    pub fn listen(&self) -> Result<()> {
236        check(unsafe { ffi::OS_SocketListen(self.0 .0) })?;
237        Ok(())
238    }
239}

241/// A TCP stream between a local and a remote socket.
242#[derive(Debug)]
243#[repr(transparent)]
244pub struct TcpStream(Socket);
245
246impl TcpStream {
247    /// Opens a TCP connection to a remote host.
248    pub fn connect(addr: SocketAddr) -> Result<Self> {
249        let mut sock_id = MaybeUninit::uninit();
250        check(unsafe {
251            ffi::OS_SocketOpen(
252                sock_id.as_mut_ptr(),
253                ffi::OS_SocketDomain_t_OS_SocketDomain_INET,
254                ffi::OS_SocketType_t_OS_SocketType_STREAM,
255            )
256        })?;
257        let sock_id = unsafe { sock_id.assume_init() };
258
259        check(unsafe { ffi::OS_SocketConnect(sock_id, &addr.0, ffi::OS_PEND as i32) })?;
260
261        Ok(TcpStream(Socket(sock_id)))
262    }
263
264    /// Reads some bytes from the stream into the specified buffer.
265    pub fn read(&mut self, buf: &mut [u8]) -> Result<usize> {
266        let bytes_read = unsafe { ffi::OS_read(self.0 .0, buf.as_mut_ptr() as *mut _, buf.len()) };
267        if bytes_read < 0 {
268            Err(Error::from(bytes_read))
269        } else {
270            Ok(bytes_read as usize)
271        }
272    }
273
274    /// Writes a buffer to the stream.
275    pub fn write(&mut self, buf: &[u8]) -> Result<usize> {
276        let bytes_written =
277            unsafe { ffi::OS_write(self.0 .0, buf.as_ptr() as *const _, buf.len()) };
278        if bytes_written < 0 {
279            Err(Error::from(bytes_written))
280        } else {
281            Ok(bytes_written as usize)
282        }
283    }
284
285    /// Accepts a new connection with an absolute timeout.
286    pub fn accept_abs(&self, abstime: OsTime) -> Result<(TcpStream, SocketAddr)> {
287        let mut remote_addr_uninit = MaybeUninit::<ffi::OS_SockAddr_t>::uninit();
288        let mut conn_sock_id = MaybeUninit::uninit();
289        check(unsafe {
290            ffi::OS_SocketAcceptAbs(
291                self.0 .0,
292                conn_sock_id.as_mut_ptr(),
293                remote_addr_uninit.as_mut_ptr(),
294                abstime.0,
295            )
296        })?;
297
298        let stream = TcpStream(Socket(unsafe { conn_sock_id.assume_init() }));
299        let remote_addr = SocketAddr(unsafe { remote_addr_uninit.assume_init() });
300        Ok((stream, remote_addr))
301    }
302
303    /// Opens a TCP connection to a remote host with an absolute timeout.
304    pub fn connect_abs(addr: SocketAddr, abstime: OsTime) -> Result<Self> {
305        let mut sock_id = MaybeUninit::uninit();
306        check(unsafe {
307            ffi::OS_SocketOpen(
308                sock_id.as_mut_ptr(),
309                ffi::OS_SocketDomain_t_OS_SocketDomain_INET,
310                ffi::OS_SocketType_t_OS_SocketType_STREAM,
311            )
312        })?;
313        let sock_id = unsafe { sock_id.assume_init() };
314
315        check(unsafe { ffi::OS_SocketConnectAbs(sock_id, &addr.0, abstime.0) })?;
316
317        Ok(TcpStream(Socket(sock_id)))
318    }

320    /// Gracefully shuts down the read, write, or both halves of the connection.
321    pub fn shutdown(&self, how: SocketShutdownMode) -> Result<()> {

344/// A set of file descriptors, for use with `select`.
345#[derive(Debug, Clone, Copy)]
346pub struct FdSet(ffi::OS_FdSet);
347
348impl FdSet {
349    /// Creates a new, empty file descriptor set.
350    pub fn new() -> Self {
351        let mut set = MaybeUninit::uninit();
352        // Should not fail.
353        unsafe { ffi::OS_SelectFdZero(set.as_mut_ptr()) };
354        Self(unsafe { set.assume_init() })
355    }
356
357    /// Clears all file descriptors from the set.
358    pub fn zero(&mut self) {
359        let _ = unsafe { ffi::OS_SelectFdZero(&mut self.0) };
360    }
361
362    /// Adds a file descriptor to the set.
363    pub fn add(&mut self, id: OsalId) {
364        let _ = unsafe { ffi::OS_SelectFdAdd(&mut self.0, id.0) };
365    }
366
367    /// Removes a file descriptor from the set.
368    pub fn clear(&mut self, id: OsalId) {
369        let _ = unsafe { ffi::OS_SelectFdClear(&mut self.0, id.0) };
370    }
371
372    /// Checks if a file descriptor is a member of the set.
373    pub fn is_set(&self, id: OsalId) -> bool {
374        unsafe { ffi::OS_SelectFdIsSet(&self.0, id.0) }
375    }
376}
377
378impl Default for FdSet {
379    fn default() -> Self {
380        Self::new()
381    }
382}

396/// Waits for events across multiple file handles with a relative timeout.
397pub fn select_multiple(
398    read_set: Option<&mut FdSet>,
399    write_set: Option<&mut FdSet>,
400    timeout_ms: i32,
401) -> Result<()> {
402    let read_ptr = read_set.map_or(core::ptr::null_mut(), |s| &mut s.0);
403    let write_ptr = write_set.map_or(core::ptr::null_mut(), |s| &mut s.0);
404    check(unsafe { ffi::OS_SelectMultiple(read_ptr, write_ptr, timeout_ms) })?;
405    Ok(())
406}

420/// Properties of an OSAL socket.
421#[derive(Debug, Clone)]
422pub struct SocketProp {
423    /// The registered name of the socket.
424    pub name: CString<{ ffi::OS_MAX_API_NAME as usize }>,
425    /// The OSAL ID of the task that created the socket.
426    pub creator: OsalId,
427}
428
429/// Finds an existing socket ID by its name.
430pub fn get_socket_id_by_name(name: &str) -> Result<OsalId> {
431    let mut c_name = CString::<{ ffi::OS_MAX_API_NAME as usize }>::new();
432    c_name
433        .extend_from_bytes(name.as_bytes())
434        .map_err(|_| Error::OsErrNameTooLong)?;
435    let mut sock_id = MaybeUninit::uninit();
436    check(unsafe { ffi::OS_SocketGetIdByName(sock_id.as_mut_ptr(), c_name.as_ptr()) })?;
437    Ok(OsalId(unsafe { sock_id.assume_init() }))
438}

440/// Retrieves information about a socket.
441pub fn get_socket_info(sock_id: OsalId) -> Result<SocketProp> {
442    let mut prop = MaybeUninit::<ffi::OS_socket_prop_t>::uninit();
443    check(unsafe { ffi::OS_SocketGetInfo(sock_id.0, prop.as_mut_ptr()) })?;
444    let prop = unsafe { prop.assume_init() };
445
446    let name_cstr = unsafe { CStr::from_ptr(prop.name.as_ptr()) };
447    let mut name = CString::new();
448    name.extend_from_bytes(name_cstr.to_bytes())
449        .map_err(|_| Error::OsErrNameTooLong)?;
450
451    Ok(SocketProp {
452        name,
453        creator: OsalId(prop.creator),
454    })
