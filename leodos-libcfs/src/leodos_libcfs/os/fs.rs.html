108impl FsHeader {
109    /// Creates and initializes a new cFE file header with the specified description and subtype.
110    ///
111    /// The description will be truncated if it is longer than the header field allows.
112    pub fn new(description: &str, subtype: u32) -> Result<Self> {
113        let mut header_uninit = MaybeUninit::<ffi::CFE_FS_Header_t>::uninit();
114
115        // CFE_FS_HDR_DESC_MAX_LEN includes the null terminator.
116        let mut c_desc = CString::<{ ffi::CFE_FS_HDR_DESC_MAX_LEN as usize }>::new();
117        // This will truncate if the description is too long, which is acceptable.
118        c_desc
119            .extend_from_bytes(description.as_bytes())
120            .map_err(|_| Error::OsFsErrPathTooLong)?;
121
122        unsafe {
123            // CFE_FS_InitHeader takes a pointer and initializes the memory it points to.
124            ffi::CFE_FS_InitHeader(header_uninit.as_mut_ptr(), c_desc.as_ptr(), subtype);
125            // Now that the C function has initialized it, we can safely assume it's initialized.
126            Ok(Self(header_uninit.assume_init()))
127        }
128    }

176/// A handle to an open directory.
177///
178/// This is an iterator that yields the entries within the directory. It is
179/// a wrapper around an `osal_id_t` that will automatically call `OS_DirectoryClose` when
180/// it goes out of scope, preventing resource leaks.
181#[derive(Debug)]
182pub struct Directory {
183    id: ffi::osal_id_t,
184}
185
186impl Directory {
187    /// Opens a directory for reading.
188    pub fn open(path: &str) -> Result<Self> {
189        let c_path = c_path_from_str(path)?;
190        let mut dir_id = MaybeUninit::uninit();
191        check(unsafe { ffi::OS_DirectoryOpen(dir_id.as_mut_ptr(), c_path.as_ptr()) })?;
192        Ok(Self {
193            id: unsafe { dir_id.assume_init() },
194        })
195    }
196
197    /// Rewinds the directory stream to the beginning.
198    pub fn rewind(&mut self) -> Result<()> {
199        check(unsafe { ffi::OS_DirectoryRewind(self.id) })?;
200        Ok(())
201    }
202}
203
204impl Iterator for Directory {
205    type Item = Result<CString<{ ffi::OS_MAX_FILE_NAME as usize }>>;
206
207    /// Reads the next directory entry.
208    ///
209    /// Returns `Some(Ok(name))` for the next entry, `None` when the directory
210    /// has been fully read, or `Some(Err(e))` if an error occurs.
211    fn next(&mut self) -> Option<Self::Item> {
212        let mut dirent = MaybeUninit::<ffi::os_dirent_t>::uninit();
213        let status = check(unsafe { ffi::OS_DirectoryRead(self.id, dirent.as_mut_ptr()) });
214
215        match status {
216            Ok(_) => {
217                let dirent = unsafe { dirent.assume_init() };
218                let c_str = unsafe { CStr::from_ptr(dirent.FileName.as_ptr()) };
219                let mut s = CString::new();
220                match s.extend_from_bytes(c_str.to_bytes()) {
221                    Ok(_) => Some(Ok(s)),
222                    Err(_) => Some(Err(Error::OsErrNameTooLong)),
223                }
224            }
225            Err(Error::OsError) => None,
226            Err(err) => Some(Err(err)),
227        }
228    }
229}

231impl Drop for Directory {

234        // OS_DirectoryClose returns an osal_status_t, which can be an error.
235        // But we ignore it in drop, as with other resources in this crate.
236        let _ = unsafe { ffi::OS_DirectoryClose(self.id) };

249/// File access modes for opening or creating a file.
250#[derive(Debug, Clone, Copy)]
251#[repr(u32)]
252pub enum AccessMode {
253    /// Open for reading only.
254    ReadOnly = ffi::OS_READ_ONLY,
255    /// Open for writing only.
256    WriteOnly = ffi::OS_WRITE_ONLY,
257    /// Open for reading and writing.
258    ReadWrite = ffi::OS_READ_WRITE,
259}

261impl File {
262    /// Returns the underlying OSAL file ID.
263    pub fn id(&self) -> OsalId {
264        self.id
265    }
266    /// Opens a file with the specified access mode.
267    ///
268    /// # Arguments
269    /// * `path`: The virtual path to the file (e.g., "/ram/my_file.txt").
270    /// * `access`: The access mode, e.g., `ffi::OS_READ_ONLY`, `ffi::OS_WRITE_ONLY`,
271    ///   or `ffi::OS_READ_WRITE`.
272    pub fn open(path: &str, access: AccessMode) -> Result<Self> {
273        let c_path = c_path_from_str(path)?;
274        let mut file_id = MaybeUninit::uninit();
275        let status = unsafe {
276            ffi::OS_OpenCreate(
277                file_id.as_mut_ptr(),
278                c_path.as_ptr(),
279                ffi::OS_file_flag_t_OS_FILE_FLAG_NONE as i32,
280                access as i32,
281            )
282        };
283        check(status)?;
284        Ok(Self {
285            id: OsalId(unsafe { file_id.assume_init() }),
286        })
287    }
288
289    /// Creates a new file, or truncates an existing one, with read/write access.
290    ///
291    /// # Arguments
292    /// * `path`: The virtual path to the file to create (e.g., "/ram/new_file.dat").
293    pub fn create(path: &str) -> Result<Self> {
294        let c_path = c_path_from_str(path)?;
295        let mut file_id = MaybeUninit::uninit();
296        let flags = ffi::OS_file_flag_t_OS_FILE_FLAG_CREATE as i32
297            | ffi::OS_file_flag_t_OS_FILE_FLAG_TRUNCATE as i32;
298
299        let status = unsafe {
300            ffi::OS_OpenCreate(
301                file_id.as_mut_ptr(),
302                c_path.as_ptr(),
303                flags,
304                ffi::OS_READ_WRITE as i32,
305            )
306        };
307        check(status)?;
308        Ok(Self {
309            id: OsalId(unsafe { file_id.assume_init() }),
310        })
311    }

313    /// Reads some bytes from the file into the specified buffer.
314    ///
315    /// Returns the number of bytes read.
316    pub fn read(&mut self, buf: &mut [u8]) -> Result<usize> {
317        let bytes_read = unsafe { ffi::OS_read(self.id.0, buf.as_mut_ptr() as *mut _, buf.len()) };
318        if bytes_read < 0 {
319            Err(Error::from(bytes_read))
320        } else {
321            Ok(bytes_read as usize)
322        }
323    }
324
325    /// Writes a buffer to the file.
326    ///
327    /// Returns the number of bytes written.
328    pub fn write(&mut self, buf: &[u8]) -> Result<usize> {
329        let bytes_written =
330            unsafe { ffi::OS_write(self.id.0, buf.as_ptr() as *const _, buf.len()) };
331        if bytes_written < 0 {
332            Err(Error::from(bytes_written))
333        } else {
334            Ok(bytes_written as usize)
335        }
336    }
337
338    /// Seeks to an offset, in bytes, in a stream.
339    ///
340    /// Returns the new position from the start of the file.
341    pub fn seek(&mut self, pos: SeekFrom) -> Result<u32> {
342        let (offset, whence) = match pos {
343            SeekFrom::Start(offset) => (offset as i32, ffi::OS_SEEK_SET),
344            SeekFrom::End(offset) => (offset, ffi::OS_SEEK_END),
345            SeekFrom::Current(offset) => (offset, ffi::OS_SEEK_CUR),
346        };
347
348        let new_pos = unsafe { ffi::OS_lseek(self.id.0, offset, whence) };
349        if new_pos < 0 {
350            Err(Error::from(new_pos))
351        } else {
352            Ok(new_pos as u32)
353        }
354    }

538/// Extracts the filename from a unix style path and filename string.
539///
540/// Returns a `&str` slice of the valid filename within the provided buffer.
541pub fn extract_filename_from_path<'a>(
542    original_path: &str,
543    filename_buf: &'a mut [u8; ffi::OS_MAX_PATH_LEN as usize],
544) -> Result<&'a str> {
545    let c_path = c_path_from_str(original_path)?;
546    let status = unsafe {
547        ffi::CFE_FS_ExtractFilenameFromPath(
548            c_path.as_ptr(),
549            filename_buf.as_mut_ptr() as *mut libc::c_char,
550        )
551    };
552    check(status)?;
553
554    let c_str = unsafe { CStr::from_ptr(filename_buf.as_ptr() as *const libc::c_char) };
555    c_str.to_str().map_err(|_| Error::InvalidString)
556}

558/// Parses a filename from an input string, applying default path and extension.
559///
560/// This function uses cFE's logic to construct a complete, platform-correct
561/// file path. If the `input_name` omits a path or extension, defaults
562/// appropriate for the `category` are applied.
563///
564/// # Arguments
565/// * `output_buf`: A buffer to store the resulting fully-qualified path.
566/// * `input_name`: The (potentially partial) filename to parse.
567/// * `category`: The `FileCategory` which determines the default path and extension.
568///
569/// # Returns
570/// On success, returns a `&str` slice of the valid, null-terminated path
571/// within `output_buf`.
572pub fn parse_input_filename<'a>(
573    output_buf: &'a mut [u8; ffi::OS_MAX_PATH_LEN as usize],
574    input_name: &str,
575    category: FileCategory,
576) -> Result<&'a str> {
577    let c_input = c_path_from_str(input_name)?;
578    let status = unsafe {
579        ffi::CFE_FS_ParseInputFileName(
580            output_buf.as_mut_ptr() as *mut libc::c_char,
581            c_input.as_ptr(),
582            output_buf.len(),
583            category as u32,
584        )
585    };
586    check(status)?;
587
588    // Find the null terminator to determine the actual length of the output string.
589    let len = output_buf.iter().position(|&b| b == 0).unwrap_or(0);
590    core::str::from_utf8(&output_buf[..len]).map_err(|_| Error::InvalidString)
591}

655/// Gets the physical drive name associated with a virtual mount point.
656pub fn get_phys_drive_name(mountpoint: &str) -> Result<CString<{ ffi::OS_MAX_PATH_LEN as usize }>> {
657    let c_mount = c_path_from_str(mountpoint)?;
658    let mut buffer = [0u8; ffi::OS_MAX_PATH_LEN as usize];
659    check(unsafe {
660        ffi::OS_FS_GetPhysDriveName(buffer.as_mut_ptr() as *mut libc::c_char, c_mount.as_ptr())
661    })?;
662    let len = buffer.iter().position(|&b| b == 0).unwrap_or(0);
663    let mut s = CString::new();
664    s.extend_from_bytes(&buffer[..len])
665        .map_err(|_| Error::OsErrNameTooLong)?;
666    Ok(s)
667}

670pub fn translate_path(
671    virtual_path: &str,
672) -> Result<CString<{ ffi::OS_MAX_LOCAL_PATH_LEN as usize }>> {
673    let c_virt = c_path_from_str(virtual_path)?;
674    let mut buffer = [0u8; ffi::OS_MAX_LOCAL_PATH_LEN as usize];
675    check(unsafe {
676        ffi::OS_TranslatePath(c_virt.as_ptr(), buffer.as_mut_ptr() as *mut libc::c_char)
677    })?;
678    let len = buffer.iter().position(|&b| b == 0).unwrap_or(0);
679    let mut s = CString::new();
680    s.extend_from_bytes(&buffer[..len])
681        .map_err(|_| Error::OsErrNameTooLong)?;
682    Ok(s)
683}
