1//! Top-level OSAL application lifecycle API.
2//!
3//! # Note
4//!
5//! These functions are typically part of the BSP/runtime and not called by
6//! individual cFS applications, but are provided for completeness and for
7//! special cases like unit testing environments.
8
9use crate::error::Result;
10use crate::ffi;
11use crate::status::check;
12
13/// Initializes the OS Abstraction Layer.
14///
15/// # C-API Mapping
16/// This is a safe wrapper for `OS_API_Init`.
17///
18/// This must be called before any other OSAL routine. It is typically handled
19/// by the cFE Main entry point.
20pub fn api_init() -> Result<()> {
21    check(unsafe { ffi::OS_API_Init() })?;
22    Ok(())
23}
24
25/// Tears down and de-initializes the OSAL.
26///
27/// # C-API Mapping
28/// This is a safe wrapper for `OS_API_Teardown`.
29///
30/// This will release all OS resources and is intended for testing or controlled
31/// shutdown scenarios.
32pub fn api_teardown() {
33    unsafe { ffi::OS_API_Teardown() };
34}
35
36/// A background thread implementation that waits for events.
37///
38/// # C-API Mapping
39/// This is a safe wrapper for `OS_IdleLoop`.
40///
41/// This is typically called by the BSP main routine after all other initialization
42/// has taken place. It waits until `application_shutdown` is called.
43pub fn idle_loop() {
44    unsafe { ffi::OS_IdleLoop() };
45}
46
47/// Deletes all resources (tasks, queues, etc.) created in OSAL.
48///
49/// # C-API Mapping
50/// This is a safe wrapper for `OS_DeleteAllObjects`.
51///
52/// This is useful for cleaning up during an orderly shutdown or for testing.
53pub fn delete_all_objects() {
54    unsafe { ffi::OS_DeleteAllObjects() };
55}
56
57/// Initiates an orderly shutdown of the OSAL application.
58///
59/// # C-API Mapping
60/// This is a safe wrapper for `OS_ApplicationShutdown`.
61///
62/// This allows the task currently blocked in `idle_loop` to wake up and return.
63pub fn application_shutdown(should_shutdown: bool) {
64    unsafe { ffi::OS_ApplicationShutdown(should_shutdown as u8) };
65}
66
67/// Exits/aborts the entire application process immediately.
68///
69/// # C-API Mapping
70/// This is a safe wrapper for `OS_ApplicationExit`.
71///
72/// This function does not return and is typically only used in non-production
73/// scenarios like unit testing.
74pub fn application_exit(status: i32) -> ! {
75    unsafe { ffi::OS_ApplicationExit(status) };
76    loop {}
77}
