11/// A type-safe, zero-cost wrapper for a cFE Task ID.
12#[derive(Debug, Clone, Copy, PartialEq, Eq)]
13pub struct TaskId(pub ffi::CFE_ES_TaskId_t);
14
15/// Type alias for a cFE child task entry point function.
16pub type TaskEntryPoint = unsafe extern "C" fn();
17
18/// A handle to a cFE child task.
19///
20/// This is a wrapper around a CFE-level task ID that will automatically call
21/// `CFE_ES_DeleteChildTask` when it goes out of scope, preventing resource leaks.
22/// It must not be used for an Application's Main Task.
23#[derive(Debug)]
24pub struct ChildTask {
25    id: TaskId,
26}
27
28impl ChildTask {
29    /// Creates a new cFE child task and starts it running.
30    ///
31    /// The new task is owned by the calling Application. The stack for the task
32    /// can be provided or allocated from the system memory heap.
33    ///
34    /// # Arguments
35    /// * `name`: A unique string to identify the task.
36    /// * `entry_point`: The function that the new task will execute. This must
37    ///   be a free function with `extern "C"` linkage.
38    /// * `stack_ptr`: A pointer to the task's stack, or `core::ptr::null_mut()`
39    ///   to have cFE allocate it.
40    /// * `stack_size`: The size of the stack to allocate for the new task.
41    /// * `priority`: The priority of the new task (0=highest, 255=lowest).
42    /// * `flags`: Reserved for future use.
43    pub fn new(
44        name: &str,
45        entry_point: TaskEntryPoint,
46        stack_ptr: *mut (),
47        stack_size: usize,
48        priority: u16,
49        flags: u32,
50    ) -> Result<Self> {
51        let mut c_name = CString::<{ ffi::OS_MAX_API_NAME as usize }>::new();
52        c_name
53            .extend_from_bytes(name.as_bytes())
54            .map_err(|_| Error::OsErrNameTooLong)?;
55
56        let mut task_id = MaybeUninit::uninit();
57        check(unsafe {
58            ffi::CFE_ES_CreateChildTask(
59                task_id.as_mut_ptr(),
60                c_name.as_ptr(),
61                Some(entry_point),
62                stack_ptr as *mut _,
63                stack_size,
64                priority,
65                flags,
66            )
67        })?;
68
69        Ok(Self {
70            id: TaskId(unsafe { task_id.assume_init() }),
71        })
72    }
73
74    /// Returns the underlying `TaskId` for this child task.
75    pub fn id(&self) -> TaskId {
76        self.id
77    }
78}
79
80impl Drop for ChildTask {
81    /// Deletes the cFE child task when the `ChildTask` object goes out of scope.
82    fn drop(&mut self) {
83        // CFE_ES_DeleteChildTask can return an error, but we ignore it in drop.
84        let _ = unsafe { ffi::CFE_ES_DeleteChildTask(self.id.0) };
85    }
86}

88/// A high-level wrapper around the FFI's `CFE_ES_TaskInfo_t`.
89#[derive(Debug, Clone)]
90pub struct TaskInfo {
91    inner: ffi::CFE_ES_TaskInfo_t,
92}
93
94impl TaskInfo {
95    /// Returns the OSAL Task ID for this task.
96    pub fn task_id(&self) -> TaskId {
97        TaskId(self.inner.TaskId)
98    }
99
100    /// Returns the parent Application ID for this task.
101    pub fn app_id(&self) -> AppId {
102        AppId(self.inner.AppId)
103    }
104
105    /// Returns the execution counter for this task.
106    pub fn execution_counter(&self) -> u32 {
107        self.inner.ExecutionCounter
108    }
109
110    /// Returns the registered name of the task.
111    pub fn name(&self) -> Result<CString<{ ffi::OS_MAX_API_NAME as usize }>> {
112        let c_str = unsafe { CStr::from_ptr(self.inner.TaskName.as_ptr()) };
113        let mut s = CString::new();
114        s.extend_from_bytes(c_str.to_bytes())
115            .map_err(|_| Error::OsErrNameTooLong)?;
116        Ok(s)
117    }
118
119    /// Returns the registered name of the parent application.
120    pub fn app_name(&self) -> Result<CString<{ ffi::OS_MAX_API_NAME as usize }>> {
121        let c_str = unsafe { CStr::from_ptr(self.inner.AppName.as_ptr()) };
122        let mut s = CString::new();
123        s.extend_from_bytes(c_str.to_bytes())
124            .map_err(|_| Error::OsErrNameTooLong)?;
125        Ok(s)
126    }
127}
128
129impl TaskId {
130    /// Retrieves detailed information about the task with this ID.
131    pub fn info(&self) -> Result<TaskInfo> {
132        let mut task_info_uninit = MaybeUninit::uninit();
133        check(unsafe { ffi::CFE_ES_GetTaskInfo(task_info_uninit.as_mut_ptr(), self.0) })?;
134        Ok(TaskInfo {
135            inner: unsafe { task_info_uninit.assume_init() },
136        })
137    }
138
139    /// Deletes a child task with this ID.
140    ///
141    /// This function is a standalone wrapper for `CFE_ES_DeleteChildTask`. Using the
142    /// `ChildTask` RAII struct is generally preferred to ensure the task is always deleted.
143    /// It must not be called for an Application's Main Task.
144    pub fn delete(&self) -> Result<()> {
145        check(unsafe { ffi::CFE_ES_DeleteChildTask(self.0) })?;
146        Ok(())
147    }
148
149    /// Retrieves the cFE Task Name for this task ID.
150    pub fn name(&self) -> Result<CString<{ ffi::OS_MAX_API_NAME as usize }>> {
151        let mut buffer = [0u8; ffi::OS_MAX_API_NAME as usize];
152        check(unsafe {
153            ffi::CFE_ES_GetTaskName(
154                buffer.as_mut_ptr() as *mut libc::c_char,
155                self.0,
156                buffer.len(),
157            )
158        })?;
159
160        // Find the null terminator to determine the actual length.
161        let len = buffer.iter().position(|&b| b == 0).unwrap_or(0);
162        let mut s = CString::new();
163        s.extend_from_bytes(&buffer[..len])
164            .map_err(|_| Error::OsErrNameTooLong)?;
165        Ok(s)
166    }
167
168    /// Converts this CFE Task ID into a zero-based integer suitable for array indexing.
169    pub fn to_index(&self) -> Result<u32> {
170        let mut index = MaybeUninit::uninit();
171        check(unsafe { ffi::CFE_ES_TaskID_ToIndex(self.0, index.as_mut_ptr()) })?;
172        Ok(unsafe { index.assume_init() })
173    }
174
175    /// Retrieves the cFE Task ID for a given task name.
176    pub fn from_name(name: &str) -> Result<Self> {
177        let mut c_name = CString::<{ ffi::OS_MAX_API_NAME as usize }>::new();
178        c_name
179            .extend_from_bytes(name.as_bytes())
180            .map_err(|_| Error::OsErrNameTooLong)?;
181
182        let mut task_id = MaybeUninit::uninit();
183        check(unsafe { ffi::CFE_ES_GetTaskIDByName(task_id.as_mut_ptr(), c_name.as_ptr()) })?;
184        Ok(TaskId(unsafe { task_id.assume_init() }))
185    }
186
187    /// Retrieves the CFE Task ID of the currently executing task.
188    pub fn current() -> Result<Self> {
189        let mut task_id = MaybeUninit::uninit();
190        check(unsafe { ffi::CFE_ES_GetTaskID(task_id.as_mut_ptr()) })?;
191        Ok(TaskId(unsafe { task_id.assume_init() }))
192    }
193}

195/// Exits the calling child task.
196///
197/// This function terminates the currently running child task and does not return.
198/// It must not be called from an Application's Main Task.
199pub fn exit_child_task() -> ! {
200    unsafe {
201        ffi::CFE_ES_ExitChildTask();
202    }
203    // This function never returns, but we add a loop to satisfy the compiler.
204    loop {}
205}

207/// Increments the execution counter for the calling task.
208///
209/// This is typically not needed for main application tasks that call `run_cycle`
210/// (via `CFE_ES_RunLoop`), as the counter is incremented automatically.
211/// It is useful for child tasks or other contexts where the counter needs to
212/// be manually managed to indicate liveness.
213pub fn increment_task_counter() {
214    unsafe {
215        ffi::CFE_ES_IncrementTaskCounter();
216    }
