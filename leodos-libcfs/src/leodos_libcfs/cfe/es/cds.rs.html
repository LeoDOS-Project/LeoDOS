1//! Safe, idiomatic wrappers for the CFE Critical Data Store (CDS) API.
2//!
3//! This module provides a generic, type-safe `CdsBlock<T>` for persisting
4//! application data across resets.
5
6use crate::error::Error;
7use crate::error::Result;
8use crate::ffi;
9use crate::status;
10use crate::status::check;
11use core::marker::PhantomData;
12use core::mem;
13use core::mem::MaybeUninit;
14use heapless::CString;
15use heapless::String;
16
17/// A type-safe, zero-cost wrapper for a cFE Critical Data Store handle.
18#[derive(Debug, Clone, Copy, PartialEq, Eq)]
19#[repr(transparent)]
20pub struct CdsHandle(pub ffi::CFE_ES_CDSHandle_t);
21
22impl CdsHandle {
23    /// Retrieves the full name ("AppName.CDSName") for this CDS handle.
24    pub fn name(&self) -> Result<String<{ ffi::CFE_MISSION_ES_CDS_MAX_FULL_NAME_LEN as usize }>> {
25        let mut buffer = [0u8; ffi::CFE_MISSION_ES_CDS_MAX_FULL_NAME_LEN as usize];
26        check(unsafe {
27            ffi::CFE_ES_GetCDSBlockName(
28                buffer.as_mut_ptr() as *mut libc::c_char,
29                self.0,
30                buffer.len(),
31            )
32        })?;
33        let len = buffer.iter().position(|&b| b == 0).unwrap_or(buffer.len());
34        let vec = heapless::Vec::from_slice(&buffer[..len]).map_err(|_| Error::OsErrNameTooLong)?;
35        let str = String::from_utf8(vec).map_err(|_| Error::OsErrNameTooLong)?;
36        Ok(str)
37    }
38}

40/// Information about the status of a CDS block upon registration.
41#[derive(Debug, Clone, Copy, PartialEq, Eq)]
42pub enum CdsInfo {
43    /// The CDS block was newly created and its contents are uninitialized.
44    Created,
45    /// The CDS block already existed and its contents have been restored.
46    Restored,
47}
48
49/// A handle to a block of data in the cFE Critical Data Store.
50///
51/// This struct is generic over the data type `T` to be stored. The underlying
52/// CDS block is registered when `new` is called and persists for the life of the
53/// cFE instance (across resets).
54///
55/// The type `T` must be `Copy` and `Sized`, as the underlying CFE API performs
56/// a raw byte copy of the data.
57#[derive(Debug)]
58pub struct CdsBlock<T: Copy + Sized> {
59    handle: ffi::CFE_ES_CDSHandle_t,
60    _phantom: PhantomData<T>,
61}
62
63impl<T: Copy + Sized> CdsBlock<T> {
64    /// Registers a new CDS block with the given name or retrieves an existing one.
65    ///
66    /// This function will attempt to create a CDS block of `size_of::<T>()`.
67    ///
68    /// # Return Value
69    ///
70    /// On success, returns `Ok((CdsBlock, CdsInfo))`. The `CdsInfo` indicates
71    /// whether the block was newly created or was restored from a previous run.
72    ///
73    /// - If `CdsInfo::Created`, the application is responsible for initializing
74    ///   the data by calling `store()`.
75    /// - If `CdsInfo::Restored`, the application can immediately call `restore()`
76    ///   to retrieve the previous state.
77    ///
78    /// # Arguments
79    /// * `name`: A unique, application-local name for the CDS block.
80    pub fn new(name: &str) -> Result<(Self, CdsInfo)> {
81        let mut c_name = CString::<{ ffi::CFE_MISSION_ES_CDS_MAX_NAME_LENGTH as usize }>::new();
82        c_name
83            .extend_from_bytes(name.as_bytes())
84            .map_err(|_| Error::CfeEsCdsInvalidName)?;
85
86        let mut handle = MaybeUninit::uninit();
87        let status = unsafe {
88            ffi::CFE_ES_RegisterCDS(handle.as_mut_ptr(), mem::size_of::<T>(), c_name.as_ptr())
89        };
90
91        // check() handles true errors. We need to handle the special success cases.
92        match check(status) {
93            Ok(status::Status::EsCdsAlreadyExists) => Ok((
94                Self {
95                    handle: unsafe { handle.assume_init() },
96                    _phantom: PhantomData,
97                },
98                CdsInfo::Restored,
99            )),
100            Ok(_) => {
101                // Any other Ok status, including Status::Success, means it was created.
102                Ok((
103                    Self {
104                        handle: unsafe { handle.assume_init() },
105                        _phantom: PhantomData,
106                    },
107                    CdsInfo::Created,
108                ))
109            }
110            Err(e) => Err(e),
111        }
112    }
113
114    /// Stores a copy of `data` into the CDS block.
115    ///
116    /// This should be called after `new` reports `CdsInfo::Created`, or any
117    /// time the application wishes to update the persistent state.
118    pub fn store(&self, data: &T) -> Result<()> {
119        check(unsafe { ffi::CFE_ES_CopyToCDS(self.handle, data as *const T as *const _) })?;
120        Ok(())
121    }
122
123    /// Restores the contents of the CDS block into a new instance of `T`.
124    ///
125    /// This will perform a raw byte copy from the CDS into the returned struct.
126    /// It is safe because `T` is constrained to be `Copy`.
127    ///
128    /// Returns `Error::EsCdsBlockCrcErr` if the data in the CDS has been corrupted.
129    /// In this case, the (corrupted) data is still copied, and the application
130    /// must decide how to proceed.
131    pub fn restore(&self) -> Result<T> {
132        let mut data = MaybeUninit::<T>::uninit();
133        let status =
134            unsafe { ffi::CFE_ES_RestoreFromCDS(data.as_mut_ptr() as *mut _, self.handle) };
135        check(status)?;
136        Ok(unsafe { data.assume_init() })
137    }
138
139    /// Returns the underlying CDS handle.
140    pub fn handle(&self) -> CdsHandle {
141        CdsHandle(self.handle)
142    }
143
144    /// Finds an existing CDS Block ID by its full name ("AppName.CDSName").
145    pub fn get_id_by_name(name: &str) -> Result<CdsHandle> {
146        let mut c_name = CString::<{ ffi::CFE_MISSION_ES_CDS_MAX_FULL_NAME_LEN as usize }>::new();
147        c_name
148            .extend_from_bytes(name.as_bytes())
149            .map_err(|_| Error::OsErrNameTooLong)?;
150
151        let mut handle = MaybeUninit::uninit();
152        check(unsafe { ffi::CFE_ES_GetCDSBlockIDByName(handle.as_mut_ptr(), c_name.as_ptr()) })?;
153        Ok(CdsHandle(unsafe { handle.assume_init() }))
154    }
155}
