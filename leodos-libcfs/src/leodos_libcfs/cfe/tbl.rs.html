13/// A type alias for the callback function used to validate table loads.
14///
15/// The function receives a pointer to the table data to be validated. It should
16/// return `CFE_SUCCESS` if the data is valid, or a negative error code otherwise.
17pub type ValidationFn = ffi::CFE_TBL_CallbackFuncPtr_t;
18
19/// A handle to a cFE table.
20#[derive(Debug, Clone, Copy, PartialEq, Eq)]
21#[repr(transparent)]
22pub struct TableHandle(pub ffi::CFE_TBL_Handle_t);
23
24/// A handle to a cFE table, generic over the table's data type `T`.
25///
26/// This struct handles registration and unregistration of the table, providing
27/// a safe, RAII-based interface.
28#[derive(Debug)]
29pub struct Table<T: Sized> {
30    handle: TableHandle,
31    is_owner: bool,
32    _phantom: PhantomData<T>,
33}

35use bitflags::bitflags;

37bitflags! {
38    /// Options for table registration.
39    pub struct TableOptions: u16 {
40        /// Default table options (single-buffered, load/dump enabled).
41        const DEFAULT         = ffi::CFE_TBL_OPT_DEFAULT as u16;
42        /// The table will use a single buffer. Updates are copied from a shared working buffer.
43        const SINGLE_BUFFERED = ffi::CFE_TBL_OPT_SNGL_BUFFER as u16;
44        /// The table will have two dedicated buffers (active and inactive) for faster updates.
45        const DOUBLE_BUFFERED = ffi::CFE_TBL_OPT_DBL_BUFFER as u16;
46        /// The table's contents can be dumped but not loaded via Table Services commands.
47        const DUMP_ONLY       = ffi::CFE_TBL_OPT_DUMP_ONLY as u16;
48        /// The table is critical and its contents will be preserved in the Critical Data Store (CDS).
49        const CRITICAL        = ffi::CFE_TBL_OPT_CRITICAL as u16;
50    }
51}

56impl<T: Sized> Table<T> {
57    /// Registers a new table with cFE Table Services.
58    ///
59    /// # Arguments
60    /// * `name`: The application-local name for the table.
61    /// * `options`: Bitwise-ORed flags for table options (e.g., `TableOptions::DEFAULT`).
62    /// * `validation_fn`: An optional callback function to validate table loads.
63    pub fn new(name: &str, options: TableOptions, validation_fn: ValidationFn) -> Result<Self> {
64        let mut handle = MaybeUninit::uninit();
65        let mut c_name = CString::<{ ffi::CFE_MISSION_TBL_MAX_NAME_LENGTH as usize }>::new();
66        c_name
67            .extend_from_bytes(name.as_bytes())
68            .map_err(|_| Error::CfeTblErrInvalidName)?;
69
70        let status = unsafe {
71            ffi::CFE_TBL_Register(
72                handle.as_mut_ptr(),
73                c_name.as_ptr(),
74                size_of::<T>(),
75                options.bits(),
76                validation_fn,
77            )
78        };
79        check(status)?;
80
81        Ok(Self {
82            handle: TableHandle(unsafe { handle.assume_init() }),
83            is_owner: true,
84            _phantom: PhantomData,
85        })
86    }
87
88    /// Obtains a handle to a table registered by another application.
89    ///
90    /// This does not take ownership of the table. When this `Table` instance is dropped,
91    /// it only releases the shared handle; it does not unregister the table itself.
92    ///
93    /// # Arguments
94    /// * `name`: The full name of the table, in the format "AppName.TableName".
95    pub fn share(name: &str) -> Result<Self> {
96        let mut handle = MaybeUninit::uninit();
97        let mut c_name = CString::<{ ffi::CFE_MISSION_TBL_MAX_FULL_NAME_LEN as usize }>::new();
98        c_name
99            .extend_from_bytes(name.as_bytes())
100            .map_err(|_| Error::CfeTblErrInvalidName)?;
101
102        let status = unsafe { ffi::CFE_TBL_Share(handle.as_mut_ptr(), c_name.as_ptr()) };
103        check(status)?;
104
105        Ok(Self {
106            handle: TableHandle(unsafe { handle.assume_init() }),
107            is_owner: false,
108            _phantom: PhantomData,
109        })
110    }
111
112    /// Loads data into the table from a file.
113    pub fn load_from_file(&self, filename: &str) -> Result<()> {
114        let mut c_filename = CString::<{ ffi::OS_MAX_PATH_LEN as usize }>::new();
115        c_filename
116            .extend_from_bytes(filename.as_bytes())
117            .map_err(|_| Error::CfeTblErrFilenameTooLong)?;
118        let status = unsafe {
119            ffi::CFE_TBL_Load(
120                self.handle.0,
121                ffi::CFE_TBL_SrcEnum_CFE_TBL_SRC_FILE,
122                c_filename.as_ptr() as *const _,
123            )
124        };
125        check(status)?;
126        Ok(())
127    }
128
129    /// Loads data into the table from a memory slice.
130    pub fn load_from_slice(&self, data: &[T]) -> Result<()> {
131        let status = unsafe {
132            ffi::CFE_TBL_Load(
133                self.handle.0,
134                ffi::CFE_TBL_SrcEnum_CFE_TBL_SRC_ADDRESS,
135                data.as_ptr() as *const _,
136            )
137        };
138        check(status)?;
139        Ok(())
140    }
141
142    /// Performs periodic processing for the table (update, validate, dump).
143    /// This should be called once per application cycle for each owned table.
144    pub fn manage(&self) -> Result<()> {
145        check(unsafe { ffi::CFE_TBL_Manage(self.handle.0) })?;
146        Ok(())
147    }
148
149    /// Notifies Table Services that the application has modified the table's contents.
150    /// This is important for critical tables backed by the CDS.
151    pub fn modified(&self) -> Result<()> {
152        check(unsafe { ffi::CFE_TBL_Modified(self.handle.0) })?;
153        Ok(())
154    }
155
156    /// Gets a read-only accessor to the table's data.
157    /// The accessor locks the table and automatically releases it when dropped.
158    pub fn get_accessor(&self) -> Result<TableAccessor<'_, T>> {
159        TableAccessor::new(self.handle)
160    }
161
162    /// Returns the underlying `TableHandle`.
163    pub fn handle(&self) -> TableHandle {
164        self.handle
165    }
166
167    /// Obtains characteristics and information about a specified table by name.
168    ///
169    /// # Arguments
170    /// * `name`: The full name of the table, in the format "AppName.TableName".
171    pub fn get_info(name: &str) -> Result<TableInfo> {
172        let mut c_name = CString::<{ ffi::CFE_MISSION_TBL_MAX_FULL_NAME_LEN as usize }>::new();
173        c_name
174            .extend_from_bytes(name.as_bytes())
175            .map_err(|_| Error::CfeTblErrInvalidName)?;
176
177        let mut tbl_info_uninit = MaybeUninit::uninit();
178
179        check(unsafe { ffi::CFE_TBL_GetInfo(tbl_info_uninit.as_mut_ptr(), c_name.as_ptr()) })?;
180
181        Ok(TableInfo(unsafe { tbl_info_uninit.assume_init() }))
182    }
183
184    /// Obtains the current status of pending actions for a table.
185    pub fn status(&self) -> Result<status::Status> {
186        check(unsafe { ffi::CFE_TBL_GetStatus(self.handle.0) })
187    }
188
189    /// Updates the contents of the table if an update is pending.
190    pub fn update(&self) -> Result<()> {
191        check(unsafe { ffi::CFE_TBL_Update(self.handle.0) })?;
192        Ok(())
193    }
194
195    /// Validates the contents of a table if a validation is pending.
196    pub fn validate(&self) -> Result<()> {
197        check(unsafe { ffi::CFE_TBL_Validate(self.handle.0) })?;
198        Ok(())
199    }
200
201    /// Copies the contents of a Dump Only Table to a shared buffer.
202    ///
203    /// This should only be called by the table owner in response to a dump request,
204    /// typically after `manage()` returns `Ok(TblInfoDumpPending)`.
205    pub fn dump_to_buffer(&self) -> Result<()> {
206        check(unsafe { ffi::CFE_TBL_DumpToBuffer(self.handle.0) })?;
207        Ok(())
208    }
209
210    /// Instructs Table Services to send a message when this table requires management.
211    ///
212    /// This allows an application to be event-driven for table maintenance instead of
213    /// polling with `manage()`.
214    ///
215    /// # Arguments
216    /// * `msg_id`: Message ID to be used in the notification message.
217    /// * `command_code`: Command code to be placed in the secondary header.
218    /// * `parameter`: Application-defined value to be passed as a parameter in the message.
219    pub fn notify_by_message(
220        &self,
221        msg_id: MsgId,
222        command_code: u16,
223        parameter: u32,
224    ) -> Result<()> {
225        check(unsafe {
226            ffi::CFE_TBL_NotifyByMessage(self.handle.0, msg_id.0, command_code, parameter)
227        })?;
228        Ok(())
229    }
230
231    /// Gets read-only accessors for multiple tables at once.
232    ///
233    /// # Safety
234    /// The caller must ensure that the types `U` in the returned accessors match
235    /// the actual types of the tables identified by the handles.
236    pub unsafe fn get_accessors<const N: usize>(
237        handles: [TableHandle; N],
238    ) -> Result<[TableAccessor<'static, ()>; N]> {
239        let mut ptrs: [*mut c_void; N] = [core::ptr::null_mut(); N];
240        check(ffi::CFE_TBL_GetAddresses(
241            &mut (ptrs.as_mut_ptr() as *mut *mut c_void),
242            N as u16,
243            handles.as_ptr() as *const _,
244        ))?;
245
246        let mut accessors: [MaybeUninit<TableAccessor<'static, ()>>; N] =
247            MaybeUninit::uninit().assume_init();
248
249        for i in 0..N {
250            accessors[i].write(TableAccessor {
251                ptr: ptrs[i] as *const (),
252                handle: handles[i],
253                _phantom: PhantomData,
254            });
255        }
256
257        Ok(accessors.map(|a| a.assume_init()))
258    }
259}

261impl<T: Sized> Drop for Table<T> {
262    /// Unregisters the table if this instance is the owner.
263    fn drop(&mut self) {
264        // Only the original registrant should unregister the table.
265        // Shared handles are simply released without unregistering.
266        if self.is_owner {
267            let _ = unsafe { ffi::CFE_TBL_Unregister(self.handle.0) };
268        }
269    }
270}

272/// A safe RAII wrapper for accessing a CFE table's memory.
273///
274/// It acquires the table pointer on creation and automatically releases it when dropped.
275#[derive(Debug)]
276pub struct TableAccessor<'a, T: 'a> {
277    ptr: *const T,
278    handle: TableHandle,
279    _phantom: PhantomData<&'a T>,
280}
281
282impl<'a, T> TableAccessor<'a, T> {
283    /// Acquires a pointer to the table data.
284    pub fn new(handle: TableHandle) -> Result<Self> {
285        let mut ptr = core::ptr::null_mut();
286        let status = unsafe { ffi::CFE_TBL_GetAddress(&mut ptr, handle.0) };
287        // CFE_TBL_INFO_UPDATED is a success code, but indicates a change.
288        if status != ffi::CFE_SUCCESS && status != ffi::CFE_TBL_INFO_UPDATED {
289            return Err(Error::from(status));
290        }
291
292        Ok(Self {
293            ptr: ptr as *const T,
294            handle,
295            _phantom: PhantomData,
296        })
297    }
298}
299
300impl<'a, T> Deref for TableAccessor<'a, T> {
301    type Target = T;
302    fn deref(&self) -> &Self::Target {
303        unsafe { &*self.ptr }
304    }
305}
306
307impl<'a, T> Drop for TableAccessor<'a, T> {
308    fn drop(&mut self) {
309        // Automatically release the address when the accessor goes out of scope.
310        let _ = unsafe { ffi::CFE_TBL_ReleaseAddress(self.handle.0) };
311    }
